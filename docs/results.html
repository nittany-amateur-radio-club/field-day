These require some tidying so that they can be combined.

Code 
w3ya_25 <- fd_2025 |>
  dplyr::filter(Call == "W3YA") |>
  dplyr::mutate(year = 2025) |>
  dplyr::rename(`FINAL SCORE` = "Final Score",
                `TOTAL CW QSOs` = "Total CW QSOs",
                `TOTAL DIGITAL QSOs` = "Total Digital QSOs",
                `TOTAL PHONE QSOs` = "Total Phone QSOs",
                `TOTAL QSOs` = "Total QSOs",
                `BONUS POINTS` = "Bonus Points")

Bonus points

Code 
w3ya |>
  dplyr::filter(!is.na(bonus)) |>

  geom_point() +
  geom_line() +

Code 
w3ya |>
  dplyr::filter(!is.na(bonus)) |>
  dplyr::mutate(bonus_pct = 100*bonus/score) |>
  ggplot() +
  aes(x = year, y = bonus_pct) +
  geom_point() +
  geom_line() +
  scale_x_continuous(breaks = seq(2016, 2025, by = 1)) +
  ylab("Bonus points as a proportion of total points")
